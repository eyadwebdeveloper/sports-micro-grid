"); } Serial.println(); Serial.println("Connected to WiFi"); Serial.print("IP Address: "); Serial.println(WiFi.localIP()); // Define the route for the root URLserver.on("/", handleRoot); // Start the server server.begin(); Serial.println("HTTP server started"); } void loop() { // Handle client requests server.handleClient(); }
